```//%T 是一个占位符 表示数据数据对应的类型
//%T 打印数据类型
fmt.Printf("%T\n", a)
//%d 按照十进制打印数据
fmt.Printf("%d\n", a)
//%o 按照八进制打印数据
fmt.Printf("%o\n", b)
//%x %X 按照十六进制打印数据
fmt.Printf("%x\n", c)
fmt.Printf("%X\n", c)
//%b 按照二进制打印数据
fmt.Printf("%b\n", a)
//%f 表示输出一个浮点型 默认保留六位小数 会四舍五入
fmt.Printf("%.2f\n", a)
//%.2f 小数点保留两位会对第三位进行四舍五入
fmt.Printf("%.2f\n", b)
//%e %E 以科学计数法打印浮点型数据
fmt.Printf("%e\n", e)
fmt.Printf("%E\n", e)
//%c 按照字符打印输出
fmt.Printf("%c\n",ch)
//%s 表示字符串输出
fmt.Printf("%s\n",str)
```
